(function() {

    let antiLag = true;

    document.addEventListener("keydown", e => {
        if (e.key.toLowerCase() === "b") {
            antiLag = !antiLag;
            console.log("AntiLag =", antiLag ? "ON" : "OFF");
        }
    });

    function waitCanvas() {
        const canvas = document.querySelector("canvas");
        if (!canvas) return requestAnimationFrame(waitCanvas);
        hook(canvas);
    }
    waitCanvas();

    function hook(canvas) {

        const ctx = canvas.getContext("2d");
        const originalDraw = ctx.drawImage;

        let last = 0;

        ctx.drawImage = function (...args) {

            if (!antiLag) {
                // ch·∫°y b√¨nh th∆∞·ªùng
                return originalDraw.apply(this, args);
            }

            const now = performance.now();

            // h·∫°n ch·∫ø s·ªë l·∫ßn v·∫Ω xu·ªëng ~80 FPS
            if (now - last < 12) return;

            last = now;
            return originalDraw.apply(this, args);
        };

        console.log("AntiLag Simple Loaded ‚Äì nh·∫•n B ƒë·ªÉ b·∫≠t/t·∫Øt");
    }

})();

function calculateRemain(text) {
    if (text.endsWith('s')) {
        var match;

        match = / in:? (\d+)s$/.exec(text);
        if (match !== null) {
            return parseInt(match[1], 10);
        }

        match = /^((\d+)m)? ?(\d+)s$/.exec(text);
        if (match !== null) {
            return (parseInt(match[2], 10) || 0) * 60 + parseInt(match[3], 10); // "x || 0" is "0 if x undefined"
        }
    }

    return null;
}
